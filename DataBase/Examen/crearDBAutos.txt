CREATE DATABASE Chrysler;
USE Chrysler;
CREATE TABLE Auto (clave VARCHAR(8) NOT NULL, marca VARCHAR(20) NOT NULL, tipo VARCHAR(15) NOT NULL, precio DECIMAL(8,2) NOT NULL, PRIMARY KEY(clave));
CREATE TABLE Venta (clave VARCHAR(8) NOT NULL, marca VARCHAR(20) NOT NULL, tipo VARCHAR(15) NOT NULL, precio DECIMAL(8,2) NOT NULL, plazo INT(5) NOT NULL, interes VARCHAR(15) NOT NULL, precioTotal DECIMAL(8,2) NOT NULL, mensualidad DECIMAL(7,2) NOT NULL);
